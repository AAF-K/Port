üìù SQL Queries

SELECT product_name, COUNT(*) AS total_reviews, AVG(rating) AS avg_rating, SUM(CASE WHEN sentiment = 'positive' THEN 1 ELSE 0 END) AS positive_count FROM reviews WHERE created_at >= DATE_SUB(NOW(), INTERVAL 30 DAY) GROUP BY product_name ORDER BY avg_rating DESC, total_reviews DESC LIMIT 10;
SELECT category, sentiment, COUNT(*) AS count, ROUND(COUNT(*) * 100.0 / SUM(COUNT(*)) OVER (PARTITION BY category), 2) AS percentage FROM reviews r JOIN products p ON r.product_id = p.id GROUP BY category, sentiment ORDER BY category, sentiment;
SELECT DATE_FORMAT(created_at, '%Y-%m') AS month, AVG(rating) AS avg_rating, COUNT(*) AS review_count, AVG(CASE WHEN sentiment = 'positive' THEN 1.0 ELSE 0.0 END) AS positive_rate FROM reviews WHERE created_at >= DATE_SUB(NOW(), INTERVAL 12 MONTH) GROUP BY DATE_FORMAT(created_at, '%Y-%m') ORDER BY month ASC;
